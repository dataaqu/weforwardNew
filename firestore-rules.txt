rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Blog Posts - Public can read published posts, only admin can write
    match /blogPosts/{postId} {
      // Allow public read access only to published posts
      allow read: if resource.data.published == true;
      
      // Only authenticated admin users can create/update/delete
      allow write: if request.auth != null && 
                   request.auth.token.email in [
                     'your-admin-email@gmail.com'  // Replace with your actual admin email
                   ];
    }
    
    // Admin Users Collection (if you have one)
    match /users/{userId} {
      allow read, write: if request.auth != null && 
                        request.auth.uid == userId;
    }
    
    // All other collections - Admin only
    match /{document=**} {
      allow read, write: if request.auth != null && 
                        request.auth.token.email in [
                          'your-admin-email@gmail.com'  // Replace with your actual admin email
                        ];
    }
  }
}
